#include<stdio.h>

int main()
{
    int a,b,r,M=0,N=0,m=0;
    printf("Input 2 numbers: \n");
    scanf(" %d %d",&a,&b);
    M=a;
    N=b;
    while(b!=0)
    {
    	r=a%b;
    	a=b;
    	b=r;
    }
    printf("The GCD of %d and %d is: %d\n",M,N,a); 
    
    m=M*N/a; //math thinking...
    printf("The SCM of %d and %d is: %d\n,M,N,m);
    
    return 0;
    
}
